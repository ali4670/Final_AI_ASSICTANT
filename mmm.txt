-- 4. ENSURE STAR FUNCTION IS SYNCED
CREATE OR REPLACE FUNCTION public.decrement_stars(amount int4, uid uuid)
RETURNS int4 AS $$
DECLARE final_val int4;
BEGIN
  UPDATE public.profiles SET stars_count = stars_count - amount 
  WHERE id = uid AND stars_count >= amount
  RETURNING stars_count INTO final_val;
  IF final_val IS NULL THEN RAISE EXCEPTION 'Insufficient Stars'; END IF;
  RETURN final_val;
END; $$ LANGUAGE plpgsql SECURITY DEFINER;

-- 1. ADD XP FUNCTION
CREATE OR REPLACE FUNCTION public.add_xp(user_id uuid, xp_to_add int4)
RETURNS void AS $$
BEGIN
  UPDATE public.profiles 
  SET xp = COALESCE(xp, 0) + xp_to_add,
      level = FLOOR(SQRT((COALESCE(xp, 0) + xp_to_add) / 100)) + 1
  WHERE id = user_id;
END; $$ LANGUAGE plpgsql SECURITY DEFINER;

-- 2. UPDATE STREAK FUNCTION
CREATE OR REPLACE FUNCTION public.update_streak(user_id uuid)
RETURNS void AS $$
DECLARE
  last_study timestamptz;
BEGIN
  SELECT updated_at INTO last_study FROM public.profiles WHERE id = user_id;
  
  IF last_study IS NULL OR last_study < now() - interval '24 hours' THEN
    UPDATE public.profiles SET daily_streak = 1, updated_at = now() WHERE id = user_id;
  ELSIF last_study < now() - interval '12 hours' THEN
    UPDATE public.profiles SET daily_streak = COALESCE(daily_streak, 0) + 1, updated_at = now() WHERE id = user_id;
  END IF;
END; $$
LANGUAGE plpgsql SECURITY DEFINER;

-- 3. INCREMENT STARS FUNCTION
CREATE OR REPLACE FUNCTION public.increment_stars(user_id uuid, amount int4 DEFAULT 1)
RETURNS void AS $$
BEGIN
  UPDATE public.profiles 
  SET stars_count = COALESCE(stars_count, 0) + amount 
  WHERE id = user_id;
END; $$ LANGUAGE plpgsql SECURITY DEFINER;
